37 /*
38  * This file defines the storage for options that come from the command line related to data file
39  * persistence. Many executables that can access data files directly such as mongod and certain
40  * tools use these variables, but each executable may have a different set of command line flags
41  * that allow the user to change a different subset of these options.
42  */

92 
93  // --directoryperdb
94  // Stores each database’s files in its own folder in the data directory.
95  // When applied to an existing system, the directoryPerDB option alters
96  // the storage pattern of the data directory.
97  bool directoryperdb;
98 
99  // --syncdelay
100  // Controls how much time can pass before MongoDB flushes data to the data files
101  // via an fsync operation.
102  // Do not set this value on production systems.
103  // In almost every situation, you should use the default setting.
104  static const double kMaxSyncdelaySecs;
105  AtomicDouble syncdelay; // seconds between fsyncs
106 
107  // --queryableBackupMode
108  // Puts MongoD into "read-only" mode. MongoD will not write any data to the underlying
109  // filesystem. Note that read operations may require writes. For example, a sort on a large
110  // dataset may fail if it requires spilling to disk.
111  bool readOnly;
112 
113  // --groupCollections
114  // Dictate to the storage engine that it should attempt to create new MongoDB collections from
115  // an existing underlying MongoDB database level resource if possible. This can improve
116  // workloads that rely heavily on creating many collections within a database.
117  bool groupCollections;
118 };
